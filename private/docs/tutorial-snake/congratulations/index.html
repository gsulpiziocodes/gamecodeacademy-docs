Full Code Block#

Want to double check you've followed the tutorial correctly? Here's all the code we just wrote in one place:

snake.ino 
#include <FastLED>
#define WIDTH 16#define HEIGHT 16#define NUM_LEDS WIDTH * HEIGHT
#define DATA_PIN 0#define joyX A0#define joyY A1
enum Direction {    up,    down,    left,    right};
struct Pos {    int x;    int y;};
struct Snake {    Pos positions[256];    int len;    Direction dir;};
struct Food {    Pos pos;};
Snake snake = {0};Food food = {0};CRGB leds[NUM_LEDS] = {0};
int translateMatrixIndex(int i) {    int r = i / WIDTH;    int c = i % WIDTH;    i = c * WIDTH + (15 - r);    if (i / 16 % 2 == 1)        return 2 * (7.5 - (i % 16)) + i;    return i;}
void initGame() {    // Setup Snake    snake.len = 1;    Pos p;    p.x = 8;    p.y = 8;    snake.positions[0] = p;    snake.dir = Direction::left;
    // Init food    food.pos.x = rand() % WIDTH;    food.pos.y = rand() % HEIGHT;}
Pos updatePos(Pos pos, Direction dir) {    switch (dir) {        case Direction::up: {            pos.y++;            return pos;        }        case Direction::down: {            pos.y--;            return pos;        }        case Direction::left: {            pos.x--;            return pos;        }        case Direction::right: {            pos.x++;            return pos;        }    }}
bool isValidPosition(Pos pos) {    if (pos.x < 0 || pos.x >= WIDTH || pos.y < 0 || pos.y >= HEIGHT) {        return false;    }    for (int i = 0; i < snake.len; i++) {        if (pos.x == snake.positions[i].x && pos.y == snake.positions[i].y) {            return false;        }    }    return true;}
void updateSnake() {    // Read joystick values    int xValue = analogRead(joyX);    int yValue = analogRead(joyY);    int xMap = map(xValue, 0, 1023, -10, 10);    int yMap = map(yValue, 0, 1023, -10, 10);

    // Handle movement logic    if (abs(xMap) > 3 || abs(yMap) > 3) {        if (abs(xMap) > abs(yMap)) {            if (xMap > 0) {                snake.dir = Direction::right;            } else {                snake.dir = Direction::left;            }        } else {            if (yMap > 0) {                snake.dir = Direction::down;            } else {                snake.dir = Direction::up;            }        }    }
    // Check collision    Pos newPos = updatePos(snake.positions[0], snake.dir);    if (!isValidPosition(newPos)) {        initGame();        return;    }
    // Check food    bool didEat = newPos.x == food.pos.x && newPos.y == food.pos.y;    if (didEat) { // Move food        food.pos.x = rand() % WIDTH;        food.pos.y = rand() % HEIGHT;        snake.len++;    }
    // Move snake    Pos newPositions[256];    newPositions[0] = newPos;    for (int i = 1; i < snake.len; i++) {        newPositions[i] = snake.positions[i - 1];    }    memcpy(snake.positions, newPositions, sizeof snake.positions);
    // Move food if it spawned on snake    while (!isValidPosition(food.pos)) {        food.pos.x = rand() % WIDTH;        food.pos.y = rand() % HEIGHT;    }}
void renderGame() {    // Zero LED display    memset(leds, 0, sizeof leds);    // Render snake    for (int i = 0; i < snake.len; i++) {        int index = snake.positions[i].y * WIDTH + snake.positions[i].x;        leds[translateMatrixIndex(index)] = 0x44ff44;    }    // Render food    int index = food.pos.y * WIDTH + food.pos.x;    leds[translateMatrixIndex(index)] = 0xff0000;    // Draw LEDs    FastLED.show();}
void setup() {    FastLED.setBrightness(20);    FastLED.addLeds<WS2812B, DATA_PIN, GRB>(leds, NUM_LEDS);
    initGame();}
void loop() {    // Update game    updateSnake();
    // Render Game to LEDs    renderGame();
    // Wait for next tick    delay(500);}
